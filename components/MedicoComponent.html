import { Component, OnInit } from '@angular/core';
import { FormGroup, FormControl, Validators } from '@angular/forms';
import { MedicoService } from '../../services/medico/medico.service';
import { Router } from '@angular/router';
import { Medico } from '../../models/medico.model';
import { EspecialidadService } from '../../services/especialidad/especialidad.service';
import { Especialidad } from '../../models/especialidad.model';



declare let $: any;
declare let swal: any;

@Component({
  selector: 'app-medico',
  templateUrl: './medico.component.html',
  styles: []
})
export class MedicoComponent implements OnInit {

  forma: FormGroup;
  desde: number = 0;
  equalCedula : any = {};
  abrirModal = false;
  especialidades: Especialidad[] = [];
  totalRegistro:number = 0;

  showEspecialidades = [];

  especialidadMedico : any[] = [];

  /**
   * Constructor
   * @param _medicoService  Inyección del servicio de medicos.
   * @param router
   * @param _especialidadService Inyección del servicio de especialidades.
   */
  constructor(public _medicoService: MedicoService,
    public router: Router,
    public _especialidadService: EspecialidadService) {

     }

  ngOnInit() {
    this.forma = new FormGroup({
      registro: new FormControl("RN1", Validators.required),
      citas: new FormControl(5, Validators.required),
      nombres: new FormControl("Vanessa Nicole", Validators.required),
      apellidos: new FormControl("Iñiguez Gualan", Validators.required),
      cedula: new FormControl("0951342112", Validators.required),
      edad: new FormControl(20, Validators.required),
      direccion: new FormControl("La Paz", Validators.required),
      genero: new FormControl("Femenino", Validators.required),
      telefono: new FormControl("0981991598", Validators.required),
      sueldo: new FormControl(1500, Validators.required)
   });
     this.forma.controls['cedula']
              .valueChanges
              .subscribe(data =>{
                if(data.length === 10){
                  this._medicoService.verifyCedula(data).subscribe( (response : any)=>{
                    this.equalCedula = response.data;
                    console.log(response);
                  });
                }
              });
  }


/**
 *Método para registrar un nuevo médico, llama a un método del servicio
 * @param medico Parámetro necesario para registrar un nuevo médico
 */
  registrarMedico(){

    if(this.forma.invalid) return;

    if(this.equalCedula.equal){
      swal('Error', this.equalCedula.mensaje, 'error');
      return;
    }

    let valueForm = this.forma.value;

    let genero = (valueForm.genero === "Masculino") ? true : false;

    let medico = new Medico(
      valueForm.registro,
      valueForm.citas,
      valueForm.nombres,
      valueForm.apellidos,
      valueForm.cedula,
      valueForm.edad,
      "",
      valueForm.cedula,
      valueForm.direccion,
      genero,
      valueForm.telefono,
      valueForm.sueldo
    );

    let auxMed = medico;

    auxMed.especialidades = this.extraerExternalEspecialidades();

    this._medicoService.crearMedico(medico)
              .subscribe(resp =>{
                this.router.navigate(['/medicos']);
              })

  }

  /**
   *
   * @ignore
   */
  cambiarDesde(valor: number){

    let desde = this.desde + valor;
    console.log(desde);

    if(desde >= this.totalRegistro){
      return;
    }

    if(desde < 0){
      return;
    }

    this.desde += valor;
    this.cargarEspecialidades();
  }


  /**
   * Método para cargar todas las especialidades.
   */
  cargarEspecialidades(){

    this._especialidadService.cargarEspecialidades()

    .subscribe((resp:any) =>{
      console.log('Lista de especialidades',resp);
      this.totalRegistro = resp.data.conteo;
      this.especialidades = resp.data.especialidades;

      this.showEspecialidades = [];

      this.especialidades.forEach((element, indice )=>{

        let showEspecialidad = {
          indice,
          nombre: element.nombre,
          precioConsulta: element.precioConsulta,
          external_id: element.external_id,
          estado: false
        }

        if(this.verificarEspecialidadSeleccionada(showEspecialidad)){
          showEspecialidad.estado = true;
        }
        this.showEspecialidades.push(showEspecialidad);

      });
});
}

/**
 * Método para abril el modal.
 */
  abrirModalAhora(){
    this.abrirModal = true;
    this.cargarEspecialidades();
  }

/**
 * Método para deterner el proceso de agregar especialidades.
 */
  cancelarEspecialidades() {
    this.abrirModal = false;
    this.especialidadMedico = [];
  }


  /**
   * Método para agregar especialidades a un nuevo médico.
   * @param evento
   * @param external_especialidad externalId de la especialidad.
   */

  agregarEspecialidad( evento,indice){

    if(evento.target.checked){
      this.showEspecialidades[indice].estado = true;
      this.especialidadMedico.push(this.showEspecialidades[indice]);
    }else{
      this.showEspecialidades[indice].estado = false;
      this.especialidadMedico = this.especialidadMedico.filter(especialidad => especialidad.indice !== indice);
    }

  }


/**
 * Método para extraer el externalId de la especialidad seleccionada.
 */
  extraerExternalEspecialidades(): any[]{

    let externalEspecialidades = [];

    this.especialidadMedico.forEach(especialidad =>{
      externalEspecialidades.push(especialidad.external_id);
    });
    return externalEspecialidades;
  }


  /**
   * Método para verificar si la especialidad está seleccionada,
   * @param especialidad Especialidad.
   */
  verificarEspecialidadSeleccionada(especialidad){

    if((this.especialidadMedico.filter(especialidadMed => especialidadMed.external_id === especialidad.external_id)).length > 0){
      return true;
    }
    return false;
  }


}
