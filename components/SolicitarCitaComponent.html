import { Component, OnInit } from '@angular/core';
import { MedicoService } from '../../../services/medico/medico.service';
import { Router } from '@angular/router';
import { EspecialidadService } from '../../../services/especialidad/especialidad.service';
import { Especialidad } from '../../../models/especialidad.model';
import { Medico } from '../../../models/medico.model';

/**
 * Calendario
 */

 import dayGridPlugin from '@fullcalendar/daygrid';
 import interactionPlugin from '@fullcalendar/interaction';
import { CitaService } from '../../../services/cita/cita.service';
import { LoginService } from '../../../services/login/login.service';

 declare var $: any;
 declare let swal: any;

@Component({
  selector: 'app-solicitar-cita',
  templateUrl: './solicitar-cita.component.html',
  styles: []
})
export class SolicitarCitaComponent implements OnInit {


  /**
   * Calendario
   */
  calendarPlugins = [dayGridPlugin, interactionPlugin];


  validRange = {
    start: '2017-05-01',
    end: '2017-06-01'
  }

/**
 * Variables necesarias
 */

 currentMedico : Medico;
  medicosPorEspecialidad: Medico[] = [];
  especialidades: Especialidad[] = [];
  currentEspecialidad : Especialidad;
  desde: number = 0;
  totalRegistro:number = 0;
  desdeMedicos: number = 0;
  totalMedicos: number = 0;
  showEspecialidades = [];
  especialistasListos: boolean;
  calendarListo: boolean;

  /**
   * Variables para la cita
   */

   fechaSeleccionada: string;
   horaSelecionada: string = "Seleccione una hora";
   horasDisponibles = ["10:00", "11:00", "12:00", "15:00", "16:00", "17:00", "18:00"];



/**
 * Constructor
 * @param _medicoService
 * @param router
 * @param _especialidadService
 */

  constructor(public _medicoService: MedicoService,
    public router: Router,
    public _especialidadService: EspecialidadService,
    public _citaService: CitaService,
    public _loginService: LoginService) { }

  ngOnInit() {
    this.cargarEspecialidades();
  }

  /**
   * Método que carga todas las especialidades existentes.
   */
  cargarEspecialidades(){

    this._especialidadService.cargarEspecialidades(this.desde)
                            .subscribe((resp:any) =>{
                                  this.totalRegistro = resp.data.conteo;
                                  this.especialidades = resp.data.especialidades;
                            });
}


/**
 * Este método ayuda para la paginación de especialidades.
 * @param valor
 */

cambiarDesde(valor: number){

  let desde = this.desde + valor;
  console.log(desde);

  if(desde >= this.totalRegistro){
    return;
  }

  if(desde < 0){
    return;
  }

  this.desde += valor;
  this.cargarEspecialidades();
}


/**
 * Método para mostrar todas las especialidades.
 * @param especialidad
 */
mostrarEspecialistas(especialidad: Especialidad){

  this.especialistasListos = true;
  this.currentEspecialidad = especialidad;
 this.cargarMedicosPorEspecialidad(especialidad);

}


/**
 * @ignore
 */
mostrarNuevamenteEspecialidades(){
  this.especialistasListos = false;
  this.currentEspecialidad = null;
  this.cargarEspecialidades();
}

/**
 * Método para cargar los médicos por especialidad.
 * @param especialidad
 */
cargarMedicosPorEspecialidad(especialidad){

  this._especialidadService.listarMedicosPorEspecialidad(especialidad.external_id, this.desdeMedicos).subscribe( (resp:any) =>{
    this.totalMedicos = resp.data.conteo;
    this.medicosPorEspecialidad = resp.data.medicos;
  });

}

/**
 *
 * @ignore
 */
cambiarDesdeMedicos(valor: number){
  let desde = this.desdeMedicos + valor;

  if(desde >= this.totalMedicos){
    return;
  }

  if(desde < 0){
    return;
  }

  this.desdeMedicos += valor;
  this.cargarMedicosPorEspecialidad(this.currentEspecialidad);
}


/**
 * Método para cargar la fecha para el calendario.
 */
cargarFechaCalendar(){

  this.validRange = {
    start: this.getCurrentDate('INICIO'),
    end: this.getCurrentDate('FINAL')
  }
}

/**
 * Método para convertir la fecha para el calendario.
 * @param tipo
 */
getCurrentDate(tipo: string): string{

  let fechaActual = new Date();

  let Unmes = 1000 * 60 * 60 * 24 * 30;


  if(tipo === "FINAL"){

    fechaActual.setTime(fechaActual.getTime() + Unmes);
  }


  let month = fechaActual.getMonth() + 1;

  let currentMonth = month < 10 ? `0${month}` : `${month}`;

  let date = fechaActual.getDate();

  let currentDate = date < 10 ? `0${date}` : `${date}`;

 return `${fechaActual.getFullYear()}-${currentMonth}-${currentDate}`;

}


/**
 * Método para mostrar el calendario
 * @param medicoSeleccionado
 */
mostrarCalendar(medicoSeleccionado: Medico){
  this.currentMedico = medicoSeleccionado;
  this.cargarFechaCalendar();
  this.calendarListo = true;
}



/**
 * @ignore
 */
mostrarNuevamenteEspecialistas(){
  this.calendarListo = false;
  this.cargarMedicosPorEspecialidad(this.currentEspecialidad);
}



/**
 *
 * @param arg
 */
seleccionarHorario(event){ //handler method

  console.log(event);

 this.fechaSeleccionada = event.startStr;
 this.cargarHorasDisponibles();


  $('#exampleModal').modal('show');
}

agendarCita(){
  console.log(this.horaSelecionada);
  if(this.horaSelecionada === "Seleccione una hora"){
    swal('Error', 'Por favor seleccione una hora', 'warning');
  }

  swal({
    title: "Está seguro?",
    text:
      " Esta a punto de agendar una cita el día " +
      this.fechaSeleccionada +
      " a las " + this.horaSelecionada + " con el médico " + this.currentMedico.nombres + " en la especialidad " + this.currentEspecialidad.nombre,
    icon: "warning",
    buttons: true,
    dangerMode: true
  }).then(aceptar => {


    if (aceptar) {

      let citaBody = {
        medico: this.currentMedico.external_id,
        precioConsulta: this.currentEspecialidad.precioConsulta,
        fecha: this.fechaSeleccionada,
        hora: this.horaSelecionada

      }

      this._citaService.crearCita(this._loginService.usuario.external_id, citaBody)
              .subscribe(resp =>{

                $('#exampleModal').modal('hide');
                /**Redireccionar a consultar citas*/
              });
    }
  });
}

cargarHorasDisponibles(){

  this._citaService.cargarHorasDisponibles(this.fechaSeleccionada,
     this.currentMedico.external_id).subscribe((resp:any) =>{
       this.horasDisponibles = resp.data;

          if(this.validRange.start === this.fechaSeleccionada){

            this.horasDisponibles = this.cargarHorasReales(this.horasDisponibles);

          }
      });

}


cargarHorasReales(arreglo: string[]){

  let horasFinalesDisponibles = [];

  let currentHour = new Date().getHours();

 arreglo.forEach(hora =>{

   let base = parseInt(hora.split(":")[0]);

   if(currentHour < base){

    horasFinalesDisponibles.push(hora);

   }

  });
  return horasFinalesDisponibles;

}




}
