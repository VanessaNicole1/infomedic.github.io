import { Component, OnInit } from "@angular/core";
import { Medico } from "../../models/medico.model";
import { Router } from "@angular/router";
import { ModalUploadService } from "../../components/modal-upload/modal-upload.service";
import { MedicoService } from "../../services/services.index";
import { EspecialidadService } from '../../services/especialidad/especialidad.service';
import { Especialidad } from '../../models/especialidad.model';

declare let $: any;
declare let swal: any;

@Component({
  selector: "app-medicos",
  templateUrl: "./medicos.component.html",
  styles: []
})
export class MedicosComponent implements OnInit {
  medicos: Medico[] = [];
  especialidades: Especialidad[] = [];
  desde: number = 0;
  desdeEspecialidades: number = 0;
  totalRegistro: number = 0;
  cargando: boolean = true;
  totalRegistrosEspecialidades: number = 0;

  currentMedico: Medico;

  showEspecialidades: boolean;

/**
 * Constructor
 * @param _medicoService
 * @param router
 * @param _modalUploadService
 * @param _especialidadService
 */
  constructor(
    public _medicoService: MedicoService,
    public router: Router,
    public _modalUploadService: ModalUploadService,
    public _especialidadService: EspecialidadService
  ) {}

  ngOnInit() {
    this.cargarMedicos();

    this._modalUploadService.notificacion.subscribe(resp =>
      this.cargarMedicos()
    );
  }

  /**
   *
   * @ignore
   */
  mostrarModal(external_id: string) {
    this._modalUploadService.mostrarModal("MED_ROLE", external_id);
  }



  /**
   * Método que sirve para cargar médicos.
   */
  cargarMedicos() {
    this.cargando = true;

    this._medicoService.cargarMedicos(this.desde).subscribe((resp: any) => {
      console.log(resp);
      this.totalRegistro = resp.data.conteo;
      this.medicos = resp.data.medicos;
      this.cargando = false;
    });
  }

  /**
   *
   * @ignore
   */
  cambiarDesde(valor: number) {
    let desde = this.desde + valor;
    console.log(desde);

    if (desde >= this.totalRegistro) {
      return;
    }

    if (desde < 0) {
      return;
    }

    this.desde += valor;
    this.cargarMedicos();
  }

  /**
   * Método que permite buscar un usuario en especial.
   * @param termino
   */
  buscarUsuario(termino: string) {
    if (termino.length <= 0) {
      this.cargarMedicos();
      return;
    }


  }

  /**
   * Método para guardar un médico actualizado.
   */

  guardar() {
    this._medicoService.actualizarMedico(this.currentMedico).subscribe(resp => {
      this.cargarMedicos();
    });
    $("#exampleModalCenter").modal("hide");
  }

  /**
   * Método para borrar un médico en especial.
   * @param medico Recibe como parámetro un médico.
   */
  borrarMedico(medico: Medico) {
    swal({
      title: "Está seguro?",
      text: "Esta a punto de eliminar a " + medico.nombres,
      icon: "warning",
      buttons: true,
      dangerMode: true
    }).then(borrar => {
      console.log(borrar);

      if (borrar) {
        this._medicoService
          .borrarMedico(medico.external_id)
          .subscribe(borrado => {
            console.log(borrado);
            this.cargarMedicos();
          });
      }
    });
  }

  /**
   * Método para agregar un nuevo médico.
   * @param medico  Recibe como parámetro un médico.
   */
  agregarMedico(medico) {
    this.currentMedico = medico;

    this._medicoService
      .cargarMedicosConEspecialidades(this.currentMedico.external_id)
      .subscribe((resp: any) => {
        this.currentMedico = resp.data;
      });
  }

  /**
   * Método que sirve para cargar médico actual.
   */

  cargarMedicoActual() {
    this._medicoService
      .cargarMedicosConEspecialidades(this.currentMedico.external_id)
      .subscribe((resp: any) => {
        this.currentMedico = resp.data;
      });
  }

  /**
   * Método para eliminar una especialidad de un médico en especial.
   * @param especialidad Recibe como parámetro una especialidad.
   */
  borrarEspecialidad(especialidad) {
    swal({
      title: "Está seguro?",
      text:
        "Esta a punto de eliminar la especialidad" +
        especialidad.nombre +
        "del médico",
      icon: "warning",
      buttons: true,
      dangerMode: true
    }).then(borrar => {
      console.log(borrar);

      if (borrar) {
        this._medicoService
          .eliminarEspecialidad(
            this.currentMedico.external_id,
            especialidad.external_id
          )
          .subscribe(borrado => {
            this.cargarMedicoActual();
            this.cargarEspecialidadesDistintas(this.currentMedico.external_id);
          });
      }
    });
  }


  /**
   *
   * @ignore
   */
  cambiarDesdeEspecialidades(valor: number) {

    let desde = this.desdeEspecialidades + valor;


    if (desde >= this.totalRegistrosEspecialidades) {
      return;
    }

    if (desde < 0) {
      return;
    }

    this.desdeEspecialidades += valor;

    this.cargarEspecialidadesDistintas(this.currentMedico.external_id);
  }

  /**
   * Método para buscar un médico en específico.
   * @param termino Termino
   */

  buscarMedico(termino: string) {
    if (termino.length <= 0) {
      this.cargarMedicos();
      return;
    }

    this._medicoService
      .buscarMedicos(termino)
      .subscribe((medicos: Medico[]) => {
        this.medicos = medicos;
      });
  }



  /**
   *
   * @ignore
   */
  cargarEspecialidades(external_id) {
    this.showEspecialidades = true;
    this.cargarEspecialidadesDistintas(external_id);

  }


  /**
   * Método para cargar especialidades para escoger.
   * @param external_id
   */
  cargarEspecialidadesDistintas(external_id){
    this._especialidadService.cargarEspecialidadesDistintas(external_id, this.desdeEspecialidades)
    .subscribe( (resp:any) =>{
      this.totalRegistrosEspecialidades = resp.data.conteo - this.currentMedico.especialidades.length;
      this.especialidades = resp.data.especialidades;
    })
  }

/**
 * Método para agregar especialidades a un médico.
 * @param especialidad Recibe una especialidad.
 */
  agregarEspecialidadMedico(especialidad){

    swal({
      title: "Está seguro?",
      text:
        "Esta a punto de agregar la especialidad" +
        especialidad.nombre +
        " al médico " + this.currentMedico.nombres,
      icon: "warning",
      buttons: true,
      dangerMode: true
    }).then(agregar => {
      console.log(agregar);

      if (agregar) {
        this._medicoService
          .agregarEspecialidad(
            this.currentMedico.external_id,
            especialidad.external_id
          )
          .subscribe(guardado => {
            console.log(guardado);
            this.cargarMedicoActual();
            this.cargarEspecialidadesDistintas(this.currentMedico.external_id);
          });
      }
    });
  }
}
