import { Component, OnInit, NgZone } from '@angular/core';

import { Router } from '@angular/router';
import { NgForm } from '@angular/forms';
/**Servicios */
import { LoginService } from '../services/login/login.service';


/**FireAuth */
import { AngularFireAuth } from "@angular/fire/auth";
import { auth } from 'firebase';

declare function init_plugins();


@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.css']
})
export class LoginComponent implements OnInit {

  /**
   * Variable global
   */
  email: string;

  /**
   * Variable global
   */
  recuerdame: boolean = false;

  /**
   * Constructor del componente LoginComponent
   * @param router Sirve para redirigir a otras p√°ginas.
   * @param _loginService Inyectamos el servicio Login.
   * @param fireAuth Recibe fireAuth para autentificarse.
   * @param ngZone
   */
  constructor(public router: Router,
    public _loginService: LoginService,
    public fireAuth: AngularFireAuth,
    public ngZone: NgZone ) { }


    auth2 : any;

    /**
     * @ignore
     */
  ngOnInit() {

    init_plugins();
    this.email = localStorage.getItem('email') || '';
    if(this.email.length > 1){
      this.recuerdame = true;
    }

  }

  /**
   * M√©todo que permite iniciar sesi√≥n con Google Sign-in.
   */
  google(){

  this.fireAuth.auth.signInWithPopup(new auth.GoogleAuthProvider()).then((response:any) =>{
    let email = response.user.email;
    let nombre = response.user.displayName;
    let foto = response.user.photoURL;

    let infoUser = {
      email,
      nombre,
      foto
    }
    this._loginService.loginGoogle(infoUser).subscribe(response =>{

      if(response){

        this.ngZone.run(()=>{
          this.router.navigate(['/dashboard']);
        });


      }else{
        this.ngZone.run(()=>{
          swal('IMPORTANTE', ' Por ser primera vez que ingresas a nuestro sistema, es necesario registrar m√°s datos sobre usted', 'warning');
          this.router.navigate(['/register']);
        });

      }

    })

  });
  }

  /**
   * Este m√©todo sirve para ingresar al sistema por medio de
   * su correo y clave.
   * @param forma
   */

  ingresar(forma: NgForm){

    if(forma.invalid){
      return;
    }

    let valueForma = forma.value;

    let usuario = {
      correo: valueForma.email,
      password: valueForma.password
    }

    this._loginService.login(usuario,  this.recuerdame)
        .subscribe(correcto => this.router.navigate(['/dashboard']));

    console.log(forma.valid);
    console.log(forma.value);

  }

}
