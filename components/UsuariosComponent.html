import { Component, OnInit } from '@angular/core';
import { Usuario } from '../../models/usuario.model';
import { UsuarioService } from 'src/app/services/services.index';
import { Router } from '@angular/router';
import { ModalUploadService } from '../../components/modal-upload/modal-upload.service';

declare let $: any;
declare let swal: any;

@Component({
  selector: 'app-usuarios',
  templateUrl: './usuarios.component.html',
  styles: []
})
export class UsuariosComponent implements OnInit {

  usuarios:Usuario[] = [];
  desde: number = 0;
  totalRegistro:number = 0;
  cargando: boolean = true;

  currentUser: Usuario;

  /**
   * Constructor
   * @param _usuarioService  Inyección del servicio de usuarios.
   * @param router Redirigir al una pantalla.
   * @param _modalUploadService Inyección del servicio modalUpload.
   */
  constructor(public _usuarioService: UsuarioService,
    public router: Router,
    public _modalUploadService: ModalUploadService) { }

  ngOnInit() {
    this.cargarUsuarios();

    this._modalUploadService.notificacion
    .subscribe(resp => this.cargarUsuarios());

  }

  /**
   *
   * @ignore
   */
  mostrarModal(external_id: string){
    this._modalUploadService.mostrarModal('USER_ROLE', external_id);

  }


  /**
   * Este método sirve para cargar todos los usuarios.
   */
  cargarUsuarios(){

    this.cargando = true;
    this._usuarioService.cargarUsuarios( this.desde)
    .subscribe((resp:any) =>{
      this.totalRegistro = resp.data.conteo;
      this.usuarios = resp.data.personsList;
      this.cargando = false;
    });

  }

  /**
   * Método que sirve para paginar los usuarios.
   * @param valor
   */
  cambiarDesde(valor: number){

    let desde = this.desde + valor;
    if(desde >= this.totalRegistro){
      return;
    }

    if(desde < 0){
      return;
    }

    this.desde += valor;
    this.cargarUsuarios();
  }

  /**
   * Método que sirve para buscar un usuario en especial.
   * @param termino
   */
  buscarUsuario(termino: string){

    if( termino.length <= 0){
      this.cargarUsuarios();
      return;
    }

    this._usuarioService.buscarUsuarios(termino)
    .subscribe((usuarios: Usuario[])=>{
      this.usuarios = usuarios;
    })
  }


  /**
   * Método para guardar un  usuario actualizado.
   */
  guardar() {

    this._usuarioService.actualizarUsuario(this.currentUser).subscribe(resp => {});
   $('#exampleModalCenter').modal('hide');
  }

  /**
   * Este método sirve para borrar un usuario en especial.
   * @param usuario Recibe un usuario.
   */
  borrarUsuario(usuario: Usuario){
    swal({
      title: 'Está seguro?',
      text: 'Esta a punto de eliminar a ' + usuario.nombres,
      icon: 'warning',
      buttons: true,
      dangerMode: true
    }).then(borrar =>{

      if(borrar){

        this._usuarioService.borrarUsuario(usuario.external_id)
        .subscribe(borrado =>{
          this.cargarUsuarios();
        })

      }
    });

  }

}
