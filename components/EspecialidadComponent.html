import { Component, OnInit } from '@angular/core';
import { FormGroup, FormControl, Validators } from '@angular/forms';
import { Especialidad } from '../../models/especialidad.model';
import { EspecialidadService } from '../../services/services.index';

declare let $: any;
declare let swal: any;

@Component({
  selector: 'app-especialidad',
  templateUrl: './especialidad.component.html',
  styles: []
})
export class EspecialidadComponent implements OnInit {

  forma: FormGroup;
  especialidades: Especialidad[] = [];
  desde: number = 0;
  totalRegistro:number = 0;
  cargando: boolean = true;

  currentEspecialidad: Especialidad;

  /**
   * Constructor
   * @param _especialidadService Inyección del sercicio de especialidades.
   */
  constructor(public _especialidadService: EspecialidadService) { }

  ngOnInit() {
    this.cargarEspecialidades();
    this.forma = new FormGroup({
      nombre: new FormControl("Cardiología", Validators.required),
      precioConsulta: new FormControl(50, Validators.required),
      descripcion: new FormControl("Buena especialidad....", Validators.required),

    });
  }

  /**
   * Método que sirve para registrar una nueva especialidad.
   */
  registrarEspecialidad(){

    if(this.forma.invalid){
      return;
    }

    let valueForm = this.forma.value;

    let especialidad = new Especialidad(
      valueForm.nombre,
      valueForm.precioConsulta,
      valueForm.descripcion
    );

    this._especialidadService.crearEspecialidad(especialidad)
        .subscribe(resp =>{
          console.log(resp);

        })

  }

/**
 * Método para cargar todas las especialidades.
 */
  cargarEspecialidades(){

    this.cargando = true;

    this._especialidadService.cargarEspecialidades(this.desde)
            .subscribe((resp:any) =>{
                    console.log('Lista de especialidades',resp);
                    this.totalRegistro = resp.data.conteo;
                    this.especialidades = resp.data.especialidades;
                    this.cargando = false;
    });

  }

  /**
   * Método para buscar especialidades en su colección.
   * @param termino Termino para saber que tipo es.
   */
  buscarEspecialidad(termino: string){

    if( termino.length <= 0){
      this.cargarEspecialidades();
      return;
    }
    console.log(termino);
  }


  /**
   * Método que me sirve para actualizar la especialidad.
   */
  guardarEspecialidad(){
    this._especialidadService.actualizarEspecialidad(this.currentEspecialidad).subscribe(resp => {});
    $('#exampleModalLong').modal('hide');
  }


  /**
   * Método para borrar una especialidad.
   * @param especialidad Especialidad
   */
  borrarEspecialidad(especialidad: Especialidad){
    //console.log(usuario);
    swal({
      title: 'Está seguro?',
      text: 'Esta a punto de eliminar a ' + especialidad.nombre,
      icon: 'warning',
      buttons: true,
      dangerMode: true
    }).then(borrar =>{
      console.log(borrar);

      if(borrar){

        this._especialidadService.borrarEspecialidad(especialidad.external_id)
        .subscribe(borrado =>{
          console.log(borrado);
          this.cargarEspecialidades();
        })

      }
    });

  }

}
