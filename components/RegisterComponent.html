import { Component, OnInit } from "@angular/core";
import { FormGroup, FormControl, Validators } from "@angular/forms";

import swal from 'sweetalert';

/**Servicios */
import { UsuarioService } from '../services/services.index';
/** Modelos */
import { Usuario } from '../models/usuario.model';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';

declare function init_plugins();

@Component({
  selector: "app-register",
  templateUrl: "./register.component.html",
  styleUrls: ["./login.component.css"]
})
export class RegisterComponent implements OnInit {

  /**
   * Variable global
   */
  forma: FormGroup;

  /**
   * Variable global
   */
  googleuserFoto: string = null;

  /**
   * Variable global
   */
  equalEmail : any = {};

  /**
   * Variable global
   */
  equalCedula : any = {};

  /**
   * Constructor del componente registerComponent
   * @param _usuarioService Inyectando el servicio de usuarios.
   * @param router Sirve para redireccionar a una p√°gina.
   */
  constructor(public _usuarioService: UsuarioService,
    public router: Router) {}


    /**
     * Este m√©todo ayuda a comparar si las dos claves ingresadas
     * a la hora de registrar un usuarios son iguales.
     * @param password1 Par√°metro importante para verificar clave.
     * @param password2 Par√°metro importante para verificar clave.
     */
  sonIguales(password1: string, password2: string){

  return (group: FormGroup)=>{

    let pass1 = group.controls[password1].value;
    let pass2 = group.controls[password2].value;

    if( pass1 === pass2){
      return null;
    }

    return{
      sonIguales: true
    }
  }

  }


  ngOnInit() {
    init_plugins();
    this.forma = new FormGroup({
      nombres: new FormControl("Vanessa Nicole", Validators.required),
      apellidos: new FormControl("I√±iguez Gualan", Validators.required),
      cedula: new FormControl("0951342112", Validators.required),
      edad: new FormControl("20", Validators.required),
      email: new FormControl("nicolegualan@gmail.com", [Validators.required, Validators.email]),
      direccion: new FormControl("La Paz", Validators.required),
      genero: new FormControl("Femenino", Validators.required),
      telefono: new FormControl("0981991598", Validators.required),
      password: new FormControl("12345", Validators.required),
      password2: new FormControl("12345", Validators.required),
      condiciones: new FormControl(false)
    },
     {validators: this.sonIguales('password', 'password2')}  );


     this.forma.controls['email']
              .valueChanges
              .subscribe(data =>{
                if(data.length > 0 ){
                  this._usuarioService.verifyEmail(data).subscribe( (response : any)=>{
                    this.equalEmail = response.data;
                    console.log(response);
                  });
                }
              });


     this.forma.controls['cedula']
              .valueChanges
              .subscribe(data =>{
                if(data.length === 10){
                  this._usuarioService.verifyCedula(data).subscribe( (response : any)=>{
                    this.equalCedula = response.data;
                    console.log(response);
                  });
                }
              });




    this.loadCurrentUser();
    this.deleteCurrentUser();
  }



  /**
   * M√©todo que sirve para registrar un nuevo usuario al sistema
   */
  registrarUsuario(){

    if(this.forma.invalid){
      return;
    }

    if(!this.forma.value.condiciones){
      swal("Importante!", "Debe aceptar los t√©rminos y condiciones", "warning" );
      return;
    }

    if(this.equalCedula.equal){
      swal('Error', this.equalCedula.mensaje , 'error');
      return;
    }

    if(this.equalEmail.equal){
      swal('Error', this.equalEmail.mensaje , 'error');
      return;
    }

    let valueForm = this.forma.value;

    let genero = (valueForm.genero === "Masculino") ? true : false;

    let usuario = new Usuario(
      valueForm.nombres,
      valueForm.apellidos,
      valueForm.cedula,
      valueForm.edad,
      valueForm.email,
      valueForm.password,
      valueForm.direccion,
      genero,
      valueForm.telefono
    );

    if(this.googleuserFoto != null){
      usuario.foto = this.googleuserFoto;
    }

    this._usuarioService.crearUsuario(usuario)
    .subscribe( resp => {
        this.router.navigate(['/login']);
    });
  }



/**
 * M√©todo que sirve para asignar valor a las
 * variables email, nombre y googleuserFoto
 * mediante el localStorage
 */
  loadCurrentUser(){
    let email = localStorage.getItem('currentEmail');
    let nombre = localStorage.getItem('currentNombre');
    this.googleuserFoto = localStorage.getItem('currentPhoto');


    if(email && nombre){
      this.forma.controls['nombres'].setValue(nombre);
      this.forma.controls['email'].setValue(email);
    }
  }

  /**
   * Este m√©todo sirve para eliminar del localStorage los items currentEmail
   * y currentNombre
   */

  deleteCurrentUser(){
    localStorage.removeItem('currentEmail');
    localStorage.removeItem('currentNombre');
  }
}
