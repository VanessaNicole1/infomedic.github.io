import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { SettingsService } from '../settings/settings.service';
import { URL_SERVICIOS } from 'src/app/config/config';

import { map } from 'rxjs/operators';
import { LoginService } from '../login/login.service';
import { Cita } from '../../models/cita.model';

@Injectable({
  providedIn: 'root'
})
export class CitaService {

  currentCita: Cita;

  readonly CITA_URL = `${URL_SERVICIOS}/cita`;

  constructor(public http: HttpClient,
     public _settingService: SettingsService,
     public _loginService: LoginService) { }


      /**
     * Método que ayuda a establecer los headers dentro de cada petición.
     * Especificamente establece el token de la persona actual.
     */
    establecerHeaders(): any{

      let token = this._loginService.token;

      let headers = new HttpHeaders().set('token', token);

      return headers;

    }


    /**
     * Método para crear una nueva cita.
     * @param external_usuario
     * @param cita
     */
    crearCita(external_usuario: string,
      cita: any,
     ){
       let headers = this.establecerHeaders();

            let url =`${this.CITA_URL}/ingresar/${external_usuario}`;

            return this.http.post(url, cita, {headers: headers} );

    }
/**
 * Método para cargar las horas disponibles con un médico.
 * @param fecha
 * @param external_medico
 */
    cargarHorasDisponibles(fecha, external_medico){

      let headers = this.establecerHeaders();

      let url =`${this.CITA_URL}/horasDisponibles/${external_medico}`;

      let body = {
        fecha
      }

      return this.http.post(url, body, {headers: headers});
    }

/**
 * Método para cargar todas las citas de un determinado paciente.
 * @param external_paciente
 * @param desde
 */
    cargarTodasLasCitasPaciente(external_paciente, desde:number){

      let headers = this.establecerHeaders();

      let url =`${this.CITA_URL}/listarCitasPaciente/${external_paciente}?desde=${desde}`;

      return this.http.get(url, {headers: headers});
    }


    cargarTodasLasCitasMedico(external_paciente, desde:number){

      let headers = this.establecerHeaders();

      let url =`${this.CITA_URL}/listarCitasDiarias/${external_paciente}?desde=${desde}`;

      return this.http.get(url, {headers: headers});
    }


    asistenciaNula(external_cita){

      let headers = this.establecerHeaders();

      let url =`${this.CITA_URL}/asistencia/${external_cita}`;

      return this.http.put(url,null, {headers: headers});



    }
}
