import { Injectable, EventEmitter } from '@angular/core';


@Injectable({
  providedIn: 'root'
})
export class ModalUploadService {

  /**
   * Variable global
   */
  public tipo: string;
  /**
   * Variable global
   */
  public external_id: string;

   /**
   * Variable global
   */
  public oculto: string = 'oculto';

  /**
   * Evento.
   */
  public notificacion = new EventEmitter<any>();

  /**
   * Constructor del servicio
   */
  constructor() {
    console.log('modal upload service listo');
   }


   /**
    * Método que ayuda a ocultar el modal al momento de actualizar foto.
    */
   ocultarModal(){
    this.oculto = 'oculto';
    this.tipo = null;
    this.external_id = null;
   }

   /**
    * Método para mostral el modal.
    * @param tipo Tipo de imagen.
    * @param external_id External_id del usuario.
    */
   mostrarModal(tipo: string, external_id: string){

    this.oculto = '';
    this.external_id = external_id;

    this.tipo = tipo;

    this.oculto = '';
   }
}
