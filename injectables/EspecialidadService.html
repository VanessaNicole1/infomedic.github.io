import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { LoginService } from '../login/login.service';
import { SettingsService } from '../settings/settings.service';
import { Especialidad } from '../../models/especialidad.model';
import { URL_SERVICIOS } from 'src/app/config/config';

import { map } from 'rxjs/operators';

@Injectable({
  providedIn: 'root'
})
export class EspecialidadService {

  readonly ESPECIALIDAD_URL = `${URL_SERVICIOS}/especialidad`;

  /**
   * Constructor
   * @param http
   * @param _loginService
   * @param _settingService
   */
  constructor(
    public http: HttpClient,
    public _loginService: LoginService,
     public _settingService: SettingsService
  ) {
    console.log("servicio de especialidad listo");
   }


     /**
      * Método para crear una nueva especialidad, realiza petición al servidor
      * @param medico
      * Recibe como parámetro un objeto de tipo médico
      */
     crearEspecialidad(especialidad: Especialidad) {

      let url = `${this.ESPECIALIDAD_URL}/ingresar`;

      let headers = this.establecerHeaders();

      return this.http.post(url, especialidad, {headers : headers})
      .pipe(map ((resp:any)=>{

        this._settingService.mostrarMensajesSuccess('Se creó con éxito la especialidad');
      }));

    }


    /**
     * Método que ayuda a establecer los headers dentro de cada petición.
     * Especificamente establece el token de la persona actual.
     */
    establecerHeaders(): any{

      let token = this._loginService.token;

      let headers = new HttpHeaders().set('token', token);

      return headers;

    }


    /**
     * Método para cargar todas las especialidades desde el backend.
     * @param desde
     */
    cargarEspecialidades(desde: number = 0){

      let headers = this.establecerHeaders();

      let url = `${this.ESPECIALIDAD_URL}/listar?desde=${desde}`;

      return this.http.get(url, { headers: headers });

    }

    /**
     *
     * @ignore
     */
    buscarEspecialidad(termino: string){


    }



    /**
     * Método para actualizar una especialidad en específico.
     * @param especialidad Recibe una especialidad.
     */
    actualizarEspecialidad(especialidad: Especialidad) {

      let token = this._loginService.token;
      console.log(especialidad);

      let headers = new HttpHeaders().set("token", token);

      let url = `${this.ESPECIALIDAD_URL}/modificar/${especialidad.external_id}`;

      console.log(url);

      return this.http.put(url, especialidad, { headers: headers }).pipe(
        map((resp: any) => {
          let data = resp.data;
          console.log(data);
          swal("Especialidad actualizada", "Especialidad correcto", "success");
          return true;
        })
      );
    }


    /**
     *Método que llama al servicio para borrar una especialidad.
     * @param external_id
     * @returns Devuelve un boolean.
     */

    borrarEspecialidad(external_id: string): any{

      let token = this._loginService.token;

      let headers = new HttpHeaders().set("token", token);

      let url = `${this.ESPECIALIDAD_URL}/eliminar/${external_id}`;

      return this.http.put(url,null, { headers: headers })
          .pipe(map((resp:any)=>{
            swal('Especialidad eliminada', 'La especialidad fue eliminada correctamente', 'success');
            return true;
          }));
    }

/**
 * Método para cargas todas las especialidades.
 * @param desde
 * @param external_id
 */
    cargarEspecialidadesMedico(desde: number = 0, external_id){
      let headers = this.establecerHeaders();

      let url = `${this.ESPECIALIDAD_URL}/listarEspecialidades/${external_id}?desde=${desde}`;

      return this.http.get(url, { headers: headers });

    }



    /**
     * Método que permite cargar las especialidades distintas a las que tiene un médico.
     * @param external_id  ExternalId del la especialidad.
     * @param desde
     */
    cargarEspecialidadesDistintas(external_id, desde: number = 0){

      let headers = this.establecerHeaders();

      let url = `${this.ESPECIALIDAD_URL}/listarEspecialidades/${external_id}?desde=${desde}`;

      return this.http.get(url, { headers: headers });

    }


    /**
     * Método que sirve para listar todos los médicos de una determinada especialidad.
     * @param external_especialidad
     * @param desde
     */
    listarMedicosPorEspecialidad(external_especialidad, desde){


      let headers = this.establecerHeaders();

      let url = `${this.ESPECIALIDAD_URL}/listarDeterminadaEspecialidad/${external_especialidad}?desde=${desde}`;

      return this.http.get(url, { headers: headers });

    }



}
