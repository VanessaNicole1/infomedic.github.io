import { Injectable } from "@angular/core";
import { Usuario } from "../../models/usuario.model";
import { HttpClient, HttpHeaders } from "@angular/common/http";
import { URL_SERVICIOS } from "src/app/config/config";

import { map, catchError } from 'rxjs/operators';
import { LoginService } from "../login/login.service";
import { SubirArchivoService } from '../subir-Archivo/subir-archivo.service';
import { SettingsService } from '../settings/settings.service';

@Injectable({
  providedIn: "root"
})
export class UsuarioService {

  constructor(public http: HttpClient,
     public _loginService: LoginService,
     public _subirArchivo: SubirArchivoService,
     public _settingService: SettingsService) {

    console.log("servicio de usuario listo");
  }

/**
 * Método para crear un nuevo usuario.
 * @param usuario Recibe un usuario.
 */
  crearUsuario(usuario: Usuario) {
    let url = `${URL_SERVICIOS}/persona/ingresar`;

    return this.http.post(url, usuario).pipe(
      map((resp: any) => {
        this.message(usuario.correo);
        return resp;
      }),
      catchError( error =>{
        this._settingService.mostrarNormalError(error);
        throw error;
      })
    );
  }


  /**
   * Método para mostrar mensajes.
   * @param body Cuerpo del mensaje
   */
  message(body: string) {
    let mensaje = `Usuario creado ${body}, para mayor seguridad necesitamos que inicie sesión`;
    if (localStorage.getItem("currentPhoto")) {
      mensaje = localStorage.getItem("currentPhoto")
        ? `${mensaje} nuevamente`
        : mensaje;

      localStorage.removeItem("currentPhoto");
    }
    swal("Correcto", mensaje, "success");
  }



  /**
   * Método para actualizar un usuario llamando el servicio del backend.
   * @param usuario
   */
  actualizarUsuario(usuario: Usuario) {

    let token = this._loginService.token;


    let headers = new HttpHeaders().set("token", token);

    let url = `${URL_SERVICIOS}/persona/modificar/${usuario.external_id}`;



    return this.http.put(url, usuario, { headers: headers }).pipe(
      map((resp: any) => {
        let data = resp.data;

        this._loginService.guardarOnStorageBeforeLogin(data);
        swal("Usuario actualizado", "Usuario correcto", "success");
        return true;
      })
    );
  }


  /**
   * Este método sirve para cambiar la foto al usuario.
   * @param archivo
   * @param external_id
   */
  cambiarImagen(archivo: File, external_id: string ){

    let role = localStorage.getItem('role');

    this._subirArchivo.subirArchivo(archivo, role, external_id)
        .then((resp:any) =>{
          this._loginService.usuario.foto = resp.data.foto;
          swal('Imagen actualizada', 'correcto', 'success');
          this._loginService.guardarOnStorageBeforeLogin(resp.data);

        }).catch(resp =>{
          console.log(resp);
        })

  }

  /**
   * Método que sirve para cargar los usuarios.
   * @param desde
   */

  cargarUsuarios(desde: number = 0){
    let token = this._loginService.token;

    let headers = new HttpHeaders().set("token", token);

    let url = `${URL_SERVICIOS}/persona/listar?desde=${desde}`;

    return this.http.get(url, { headers: headers });

  }


  /**
   * Método que sirve para buscar determinado usuario.
   * @param termino
   */
  buscarUsuarios(termino: string){

    let url = `${URL_SERVICIOS}/busqueda/coleccion/usuarios/${termino}`;

    return this.http.get(url).pipe(map((resp:any) => resp.usuarios ));

  }


  /**
   * Método para buscar un determinado usuario con todos sus detalles.
   * @param external
   */
  buscarUnUsuario(external: string){

    let token = this._loginService.token;

    let headers = new HttpHeaders().set("token", token);

    let url = `${URL_SERVICIOS}/persona/mostrarPersona/${external}`;

    return this.http.get(url, { headers: headers });

  }

/**
 * Método para borrar un usuario en específico.
 * @param external_id
 */
  borrarUsuario(external_id: string){

    let token = this._loginService.token;

    let headers = new HttpHeaders().set("token", token);

    let url = `${URL_SERVICIOS}/persona/eliminar/${external_id}`;

    return this.http.put(url,null, { headers: headers })
        .pipe(map((resp:any)=>{
          swal('Usuario eliminado', 'El usuario fue eliminado correctamente', 'success');
          return true;
        }));
  }


/**
 * Método para verificar correo.
 * @param email
 */
  verifyEmail(email : string) {

    let url = `${URL_SERVICIOS}/persona/verifyEmail/${email}`;

    return this.http.get(url);
  }


  /**
   * Método para verificar la cédula, llamando el servicio del backend.
   * @param cedula
   */
  verifyCedula(cedula : string) {

    let url = `${URL_SERVICIOS}/persona/verifyCedula/${cedula}`;

    return this.http.get(url);
  }

/**
 * Método para cargar el historial del usuario.
 * @param external_usuario
 */
  cargarHistorialUsuario(external_usuario){

    let token = this._loginService.token;

    let headers = new HttpHeaders().set("token", token);

    let url = `${URL_SERVICIOS}/persona/obtenerHistorial/${external_usuario}`;

    return this.http.get(url, { headers: headers });
  }


  ingresarHistorialUsuario(external_usuario, historial:any){

    let token = this._loginService.token;

    let headers = new HttpHeaders().set("token", token);

    let url = `${URL_SERVICIOS}/persona/agregarHistorial/${external_usuario}`;

    return this.http.post(url, historial, { headers: headers });


  }




}
