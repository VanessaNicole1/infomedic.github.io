import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { LoginService } from '../login/login.service';
import { SubirArchivoService } from '../subir-Archivo/subir-archivo.service';
import { SettingsService } from '../settings/settings.service';
import { Medico } from '../../models/medico.model';
import { URL_SERVICIOS } from "src/app/config/config";

import { map } from 'rxjs/operators';

@Injectable({
  providedIn: 'root'
})
export class MedicoService {

  readonly MEDICO_URL = `${URL_SERVICIOS}/medico`;

  constructor(public http: HttpClient,
    public _loginService: LoginService,
     public _subirArchivo: SubirArchivoService,
     public _settingService: SettingsService
    ) {
      console.log("servicio de médico listo");
     }


     /**
      * Método para crear un nuevo médico, realiza petición al servidor
      * @param medico
      * Recibe como parámetro un objeto de tipo médico
      */
     crearMedico(medico: Medico) {

      let url = `${this.MEDICO_URL}/ingresar`;

      let headers = this.establecerHeaders();

      return this.http.post(url, medico, {headers : headers})
              .pipe(map ((resp:any)=>{

                let mensaje = `Correo: ${resp.data.correo}, la contraseña por defecto es la cédula`;

                this._settingService.mostrarMensajesSuccess(mensaje);
              }));

    }


    /**
     * Método que ayuda a establecer los headers dentro de cada petición.
     * Especificamente establece el token de la persona actual.
     */
    establecerHeaders(): any{

      let token = this._loginService.token;

      let headers = new HttpHeaders().set('token', token);

      return headers;

    }


    /**
     * Método para eliminar un médico en específico.
     * @param external_id
     */
    borrarMedico(external_id: string){

      let headers = this.establecerHeaders();


      let url = `${this.MEDICO_URL}/eliminar/${external_id}`;

      return this.http.put(url,null, { headers: headers })
          .pipe(map((resp:any)=>{
            swal('Médico eliminado', 'El médico fue eliminado correctamente', 'success');
            return true;
          }));
    }


    /**
     * Permite verificar si la cédula ya sirve en la base de datos
     * @param cedula Parámetro necesario.
     */
    verifyCedula(cedula : string) {

      let url = `${this.MEDICO_URL}/verifyCedula/${cedula}`;

      return this.http.get(url);
    }


/**
 * Método permite listar todos los médicos de la base de datos.
 * @param desde Parámetro para saber el limite que se desea recibir.
 */
  cargarMedicos(desde: number = 0){

    let headers = this.establecerHeaders();

    let url = `${this.MEDICO_URL}/listar?desde=${desde}`;

    return this.http.get(url, { headers: headers });

  }

  /**
   * Método para cargar médicos con sus especialidades.
   * @param external
   */
  cargarMedicosConEspecialidades(external){

    let headers = this.establecerHeaders();

    let url = `${this.MEDICO_URL}/listarMedico/${external}`;

    return this.http.get(url, { headers: headers });

  }


  /**
   * Método para eliminar una especialidad de un médico en específico.
   * @param external_medico
   * @param external_especialidad
   */
  eliminarEspecialidad(external_medico, external_especialidad){

    let headers = this.establecerHeaders();

    let url = `${this.MEDICO_URL}/eliminarEspecialidad/${external_medico}`;

    return this.http.put(url, {especialidad : external_especialidad}, { headers: headers });

  }


  /**
   * Método para actualizar un médico.
   * @param medico Recibe un médico.
   */
  actualizarMedico(medico: Medico) {

    let headers = this.establecerHeaders();

    let url = `${this.MEDICO_URL}/modificar/${medico.external_id}`;



    return this.http.put(url, medico, { headers: headers }).pipe(
      map((resp: any) => {

        swal("Médico actualizado", "Médico correcto", "success");
        return true;
      })
    );
  }

  /**
   * Método para buscar un determinado médico.
   * @param termino
   */
  buscarMedicos(termino: string){

    let url = `${URL_SERVICIOS}/busqueda/coleccion/medicos/${termino}`;

    return this.http.get(url).pipe(map((resp:any) => resp.medicos ));

  }

  /**
   * Método para agregar una nueva especcialidad a un médico.
   * @param external_id
   * @param external_especialidad
   */
  agregarEspecialidad(external_id, external_especialidad){

    let headers = this.establecerHeaders();

    let url = `${this.MEDICO_URL}/agregarEspecialidad/${external_id}`;

    return this.http.put(url, {especialidad:external_especialidad}, { headers: headers });

  }


}
