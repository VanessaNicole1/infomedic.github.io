import { Injectable } from "@angular/core";
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { URL_SERVICIOS } from "src/app/config/config";

import { map, catchError } from 'rxjs/operators';
import { Usuario } from '../../models/usuario.model';
import { Router } from '@angular/router';
import { SettingsService } from '../settings/settings.service';
@Injectable({
  providedIn: "root"
})
export class LoginService {



  usuario: Usuario;
  token: string;
  menu: any;
  role: string;

  /**
   * Constructor de la clase.
   *
   * @param http Recibe como parámetro el http de tipo HttpClient.
   * @param router Recibe como parámetro el router de tipo Router
   * @param _settingService Recibe como parámetro un servicio.
   */

  constructor(public http: HttpClient,
    public router: Router,
    public _settingService: SettingsService) {
    this.cargarStorage();
  }

/**
 * Método que llama al servicio para renovar el token.
 */
  renuevaToken(){
    let url = `${URL_SERVICIOS}/login/renuevatoken`;
    let headers = new HttpHeaders().set("token", this.token);
  return this.http.get(url,  { headers: headers })
        .pipe(map((resp:any) =>{

          let data = resp.data;
          this.guardarEnStorage(data);
          return true;
        }),

      catchError( error =>{
        this.router.navigate(['/login']);
        this._settingService.mostrarNormalError('No se pudo renovar el token');
        throw error;
      }));
  }

  /**
   * Método que sirve para iniciar sesión por correo y cuenta.
   * @param usuario Parámetro necesario, recibe un el correo y la clave del usuario.
   * @param recordar Parámetro necesario, recibe un boolean que indica si se desea
   * recordar el correo en el inicio de sesión.
   */

  login(usuario: { correo: string; password: string },
    recordar: boolean = false ) {
    if (recordar) {
      localStorage.setItem("email", usuario.correo);
    } else {
      localStorage.removeItem("email");
    }

    let url = `${URL_SERVICIOS}/login`;

    return this.http.post(url, usuario).pipe(
      map((resp: any) => {
        let data = resp.data;

       this.guardarEnStorage(data);
        return true;
      }),
      catchError( error =>{
        this._settingService.mostrarNormalError(error);
        throw error;
      }))
  }

  /**
   * Método para cerrar sesión en el sistema
   */

  logout(){

    this.usuario = null;
    this.token = '';
    this.menu = [];
    localStorage.removeItem('token');
    localStorage.removeItem('user');
    localStorage.removeItem('role');
    localStorage.removeItem('id');
    localStorage.removeItem('menu');
    this.router.navigate(['/login']);
  }

/**
 * Método que permite iniciar sesión por Google Sign-in
 * @param email Parámetro necesario para verificar el correo.
 */
  loginGoogle(email: any) {
    let url = `${URL_SERVICIOS}/login/google`;
    return this.http.post(url, email).pipe(
      map((resp: any) => {

        let data = resp.data;
        console.log('USUARIO', data);

        if (data.message === "YES") {
          this.guardarEnStorage(data);
          return true;
        }

        localStorage.setItem('currentEmail', data.usuario.email);
        localStorage.setItem('currentNombre', data.usuario.nombre);
        localStorage.setItem('currentPhoto', data.usuario.foto);

        return false;
      })
    );
  }

/**
 *Método que permite guardar en localStorage cuando se inicia sesión.
 * @param data Recibe como parámetro un objeto de tipo any.
 */
  guardarEnStorage(data: any){
    localStorage.setItem("role", data.user.role);
    localStorage.setItem("token", data.token);
    localStorage.setItem("id", data.user.person.external_id);
    localStorage.setItem("user", JSON.stringify(data.user.person));
    localStorage.setItem("menu", JSON.stringify(data.menu));

    this.role = data.user.role;
    this.usuario = data.user.person;
    this.token = data.token;
    this.menu = data.menu;
  }

/**
 * Método que permite guardar en el localStorage un usuario
 * a la hora de modificar.
 * @param data Recibe como parámetro un objeto de tipo any.
 */
  guardarOnStorageBeforeLogin(data: any){

    localStorage.setItem("user", JSON.stringify(data));
    this.usuario = data;
  }


  /**
   * Método que permite obtener los datos guardados en el localStorage.
   */
  cargarStorage(){
    if(localStorage.getItem('token')){
      this.token = localStorage.getItem('token');
      this.usuario = JSON.parse(localStorage.getItem('user'));
      this.role = localStorage.getItem('role');
      this.menu = JSON.parse(localStorage.getItem('menu'));
    }else{
      this.token = '';
      this.usuario = null;
      this.role = '';
      this.menu = [];
    }
  }



  /**
   * Método que ayuda a verificar si se encuentra logueado.
   */
  estaLogueado(){
    return (this.token.length > 5) ? true : false;
  }









}
